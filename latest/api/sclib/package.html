  2. package io

    io utils

    io utils

    all functions are wrapped in a Try, so errors are captured and it's easy to compose.

    Example: 

    1. scala> import sclib.io._
scala> for {
     |   fh <- file("/tmp/example")
     |   _ <- fh.writeLines("1. apple")
     |   _ <- fh.appendLines(List("2. banana", "3. cherry"))
     |   _ <- fh.append(4)
     |   _ <- fh.append('.')
     |   _ <- fh.append(" dog")
     |   content <- fh.slurp
     |   _ <- fh.delete
     | } yield content
res0: scala.util.Try[String] =
Success(1. apple
2. banana
3. cherry
4. dog)
